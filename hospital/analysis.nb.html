dat.bea <- paste0("https://apps.bea.gov/api/data?&",

                  "datasetname=NIUnderlyingDetail&",
                  "TableName=U20403&",
                  "Frequency=M&",
                  "Year=2014,2015,2016,2017,2018,2019,2020&",
                  "ResultFormat=JSON") %>%
